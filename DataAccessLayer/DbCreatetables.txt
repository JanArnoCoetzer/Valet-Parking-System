CREATE TABLE Customers (
    Customer_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Car_Stored_ID NUMBER(1) DEFAULT 0 NOT NULL,
    Full_Name VARCHAR2(50) NOT NULL,
    Telephone VARCHAR2(20) NOT NULL,
    Address VARCHAR2(100) NOT NULL,
    CONSTRAINT PK_Customers PRIMARY KEY (Customer_ID)
);

CREATE TABLE Vehicles (
    ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Registration VARCHAR2(20) NOT NULL,
    Model VARCHAR2(30) NOT NULL,
    Color VARCHAR2(20) NOT NULL,
    Status VARCHAR2(30) NOT NULL,
    CONSTRAINT PK_Vehicles PRIMARY KEY (ID)
);

CREATE TABLE Operators (
    Operator_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Full_Name VARCHAR2(50) NOT NULL,
    Date_Joined VARCHAR2(20) NOT NULL,
    Full_Address VARCHAR2(100) NOT NULL,
    Telephone VARCHAR2(20) NOT NULL,
    Email VARCHAR2(100) NOT NULL,
    Password VARCHAR2(100) NOT NULL,
    Permissions VARCHAR2(5) NOT NULL,
    CONSTRAINT PK_Operators PRIMARY KEY (Operator_ID)
);

CREATE TABLE ParkingSpaces (
    Space_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Lot_Identifier VARCHAR2(30) NOT NULL,
    Status VARCHAR2(30) NOT NULL,
    Available NUMBER(1) DEFAULT 1 NOT NULL,
    CONSTRAINT PK_ParkingSpaces PRIMARY KEY (Space_ID)
);

CREATE TABLE Bookings (
    Booking_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Customer_ID NUMBER NOT NULL,
    Booking_Operator_ID NUMBER,
    Storage_Operator_ID NUMBER,
    Retrieval_Operator_ID NUMBER,
    Handing_Over_Operator_ID NUMBER,
    Edited_Operators_IDs CLOB,
    Parking_Space_ID NUMBER,
    Vehicle_ID NUMBER,
    Date_From VARCHAR2(20) NOT NULL,
    Date_To VARCHAR2(20) NOT NULL,
    Time_To VARCHAR2(10) NOT NULL,
    Time_From VARCHAR2(10) NOT NULL,
    Status VARCHAR2(30) NOT NULL,
    CONSTRAINT PK_Bookings PRIMARY KEY (Booking_ID),

    CONSTRAINT FK_Bookings_Customers
        FOREIGN KEY (Customer_ID)
        REFERENCES Customers(Customer_ID),

    CONSTRAINT FK_Bookings_Booking_Operator
        FOREIGN KEY (Booking_Operator_ID)
        REFERENCES Operators(Operator_ID),

    CONSTRAINT FK_Bookings_Storage_Operator
        FOREIGN KEY (Storage_Operator_ID)
        REFERENCES Operators(Operator_ID),

    CONSTRAINT FK_Bookings_Retrieval_Operator
        FOREIGN KEY (Retrieval_Operator_ID)
        REFERENCES Operators(Operator_ID),

    CONSTRAINT FK_Bookings_Handing_Over_Operator
        FOREIGN KEY (Handing_Over_Operator_ID)
        REFERENCES Operators(Operator_ID),

    CONSTRAINT FK_Bookings_Parking_Spaces
        FOREIGN KEY (Parking_Space_ID)
        REFERENCES ParkingSpaces(Space_ID),

    CONSTRAINT FK_Bookings_Vehicles
        FOREIGN KEY (Vehicle_ID)
        REFERENCES Vehicles(ID)
);

INSERT INTO ParkingSpaces (Space_ID, Lot_Identifier, Status, Available)
SELECT
    ROWNUM AS Space_ID,
    building || '_' || floor_code || LPAD(space_num, 2, '0') AS Lot_Identifier,
    'Available' AS Status,
    1 AS Available
FROM (
    SELECT
        CASE
            WHEN level_num <= 50 THEN 'A'
            WHEN level_num <= 100 THEN 'B'
        END AS building,
        CASE
            WHEN MOD(level_num - 1, 150) < 50 THEN 'A'
            WHEN MOD(level_num - 1, 150) < 100 THEN 'B'
            ELSE 'C'
        END AS floor_code,
        CASE
            WHEN MOD(level_num - 1, 50) = 0 THEN 1
            ELSE MOD(level_num - 1, 50) + 1
        END AS space_num
    FROM (
        SELECT LEVEL AS level_num
        FROM dual
        CONNECT BY LEVEL <= 300
    )
);
